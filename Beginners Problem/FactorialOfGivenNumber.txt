//Factorial of given number
import java.util.*;
class Main {
  public static void main(String[] args) {
    Scanner sc=new Scanner(System.in);
    System.out.println("Enter the number :");
    int N=sc.nextInt();
    int fact=1;
    for(int i=1;i<=N;i++)
      {
        fact*=i;
      }
    System.out.print("The factorial is : "+fact);
  }
}
