/*Pattern Printing
  N=5
  1
  12
  123
  1234
  12345
*/
import java.util.*;
class Main {
  public static void main(String[] args) {
   Scanner sc=new Scanner(System.in);
    System.out.println("Enter the value of N :");
    int N=sc.nextInt();
    for(int i=1;i<=N;i++)
      {
        for(int j=1;j<=i;j++)
          {
            System.out.print(j);
          }
        System.out.println("");
      }
  }
}
